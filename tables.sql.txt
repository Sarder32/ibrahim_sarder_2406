CREATE TABLE EMPLOYEES(
EMPLOYEE_ID NUMBER(2),
FIRST_NAME VARCHAR(16),
LAST_NAME VARCHAR(16),
ADDRESS VARCHER(32),
POSITION CHAR(8),
SALARY NUMBER(8),
CELL_NO NUMBER(8),
CONSTRAINT PK_EMPLOYEE PRIMARY KEY (EMPLOYEE_ID));

CREATE TABLE CUSTOMERS(
CUSTOMER_ID NUMBERE(2),
FIRST_NAME VARCHAR(16),
LAST_NAME VARCHAR(16),
ADDRESS VARCHER(32),
CELL_NO NUMBER(8),
CITY VARCHAR(4),
CONSTRAINT PK_CUSTOMER PRIMARY KEY (CUSTOMER_ID));

CREATE TABLE ORDERS(
ORDER_ID NUMBER(2),
ORDER_TIME CHAR(4),
CUSTOMER_ID NUMBER(2),
EMPLOYEE_ID NUMBER(2),
COMMISSION NUMBER(6,2),
CONSTRAINT PK_ORDER PRIMARY KEY (ORDER_ID));


CREATE TABLE DELIVERIES(
DELIVERY_ID NUMBER(2),
ORDER_ID NUMBER(2),
CUSTOMER_ID NUMBER(2),
DELIVERY_TIME CHAR(4),
CITY CHAR(4),
CONSTRAINT PK_DELIVERY PRIMARY KEY (DELIVERY_ID));


CREATE TABLE PRODUCTS(
PRODUCT_ID NUMBER(2),
CUSTOMER_ID NUMBER(2),
EMPLOYEE_ID NUMBER(2),
ORDER_ID NUMBER(2),
TOTAL_PRICE NUMBER(4),
COMMISSION NUMBER(6,2),
CONSTRAINT PK_PRODUCT PRIMARY KEY (PRODUCT_ID));


CREATE TABLE TRANSPORTATIONS(
TRANSPORTATION_ID NUMBER(2),
DATE CHAR(4),
EMPLOYEE_ID NUMBER(2),
ORDER_ID NUMBER(2),
PRODUCT_ID NUMBER(6,2),
CONSTRAINT PK_TRANSPORTATION PRIMARY KEY (TRANSPORTATION_ID));
